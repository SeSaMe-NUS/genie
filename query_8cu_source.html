197 
198  if (table.build_status() == inv_table::not_builded)

204

206  u32 local_threshold;
207 
208  for (auto di = _attr_map.begin(); di != _attr_map.end(); ++di)
209  {

211  int index = di->first;
212 
213  for (unsigned int i = 0; i < ranges->size(); ++i)
214  {
215  range& d = ranges->at(i);
216  if (d.selectivity > 0)
217  {
218  local_threshold = ceil(d.selectivity * table.i_size());

227  return;

234  continue;
235  count += table.get_posting_list_size(index, vi);
