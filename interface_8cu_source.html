917 void GPUGenie::sequence_to_gram(vector<vector<int> > & sequences, vector<vector<int> >& gram_data,
918  int max_value, int gram_length)
919 {
920  int num_of_value = max_value + 1;
921  int gram_value = 0;
922 
923  for(unsigned int i = 0 ; i < sequences.size() ; ++i)
924  {
925  int max_index = sequences[i].size() - gram_length + 1;
926  if(max_index <= 0)
927  {
928  vector<int> line_null;
929  vector<int> temp_line;
930  for(unsigned int p = 0; p < (unsigned int)gram_length; ++p)
931  {
932  if(p<sequences[i].size() <= 0) break;
933  if(p < sequences[i].size())
934  line_null.push_back(sequences[i][p]);
935  else
936  line_null.push_back(sequences[i][0]);
937 
938  }
939  int number = 0;
940  for(unsigned int p = 0; p < (unsigned int)gram_length; ++p)
941  {
942  number = number*num_of_value + line_null[p];
943  }
944  temp_line.push_back(number);
945  gram_data.push_back(temp_line);
946  continue;
947  }
948  vector<int> line;
949  for(unsigned int j = 0 ; j < (unsigned int)max_index ; ++j)
950  {
951  gram_value = 0;
952  for(unsigned int k = 0 ; k < (unsigned int)gram_length ; ++k )
953  {
954  gram_value = gram_value*num_of_value + sequences[i][j + k];
955  }
956  line.push_back(gram_value);
957  }
958  gram_data.push_back(line);
959  }
960 }

971  if(data[i][j] < min_value) min_value = data[i][j];

973  for(unsigned int i = 0 ; i < data.size() ; ++i)
974  {
975  vector<int> line;

977  line.push_back(data[i][j] - min_value);
