260 __device__ int
261 GPUGenie::DeviceVarintCodec::numIntsStartingHere(uint32_t *d_in, int idxUnpack, int comprLength)
262 {
263  // This function checks the last byte of the preceding uint32_t and the first 3 bytes of the current uint32_t, i.e.
264  // d_in[idxUnpack]. If such byte value has 1 in the highest bit, then a new int must start in this uint32_t
265  uint8_t* nextBytePtr = reinterpret_cast<uint8_t*>(d_in + idxUnpack);
266  uint8_t prevByte = idxUnpack > 0 ? (d_in[idxUnpack-1] >> 24) : 0xFF;
267  int numIntsStartingHere = 0;
268 
269  for (int i = 0; i < 4; i++)
270  {
271  if (prevByte & 128)
272  numIntsStartingHere++;
273 
274  prevByte = *nextBytePtr;
275  nextBytePtr++;
276  }
277  // the last compressed uint32_t may have leading bits of some bytes set to 0, even though no integer starts there
278  if (idxUnpack == comprLength - 1 && !(prevByte & 128))
279  numIntsStartingHere--;
280  return numIntsStartingHere;
281 }
